import RoadmapColumns from '/src/components/roadmap-columns.mdx';

export function getYearQuarters(year) {
    const nextQuarters = [];

    // Loop to get the next 3 quarters
    for (let i = 1; i < 5; i++) {
        nextQuarters.push({
            quarter: i,
            year: year,
            header: `Q${i} - ${year}`
        });
    }

    return nextQuarters;
}

<RoadmapColumns columns={getYearQuarters(props.year)}></RoadmapColumns>
