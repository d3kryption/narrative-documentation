import RoadmapCard from '/src/components/roadmap-card.mdx'

export function GetCards({data}) {
	let cards = [];

    data.forEach((item, index) => {
		cards.push(
			<RoadmapCard name={item.name} desc={item.desc} type={item.type} key={index}></RoadmapCard>
		)
	});

	return cards;
}

<GetCards data={props.data}></GetCards>
